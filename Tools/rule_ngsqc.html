ngsqc.rl - A Snakemake rule that runs NGS-QC Toolkit to determine NGS sequence quality.
Version: x.x

# - ngsqc.rl
# FILE: ngsqc.rl
# Date or version: XXXX
rule ngsqc:
      input:
         "{x}.R1.fastq",
         "{x}.R2.fastq"
      output:
         "QC/{x}.R1.fastq_filtered",
         "QC/{x}.R2.fastq_filtered"
      params: illuqc=config['bin']['NGSQCTOOLKIT']
shell: "{params.illuqc} -pe {input[0]} {input[1]} 3 A -o QC"
